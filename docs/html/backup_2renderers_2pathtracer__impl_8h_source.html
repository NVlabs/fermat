118  fprintf(stderr, " glossy : %u\n", m_options.glossy_scattering ? 1 : 0);

167  m_mesh_vtls->init(n_light_paths, renderer, 0u );

193 void PathTracer::render(const uint32 instance, RenderingContext& renderer)
194 {
196  // pre-multiply the previous frame for blending
197  renderer.rescale_frame( instance );
198 
199  //fprintf(stderr, "render started (%u)\n", instance);
200  const uint2 res = renderer.res();
201  const uint32 n_pixels = res.x * res.y;
202 
203  cugar::memory_arena arena( m_memory_pool.ptr() );
204 
205  PTRayQueue in_queue;
206  PTRayQueue scatter_queue;
207  PTRayQueue shadow_queue;
208 
209  alloc_queues(
210  m_options,
211  n_pixels,
212  in_queue,
213  scatter_queue,
214  shadow_queue,
215  arena );
216 
217  // fetch a view of the renderer
218  RenderingContextView renderer_view = renderer.view(instance);
219 
221  // instantiate the vertex processor
222  PTVertexProcessor vertex_processor;
225 
226  // alloc space for device timers
227  uint64* device_timers = arena.alloc<uint64>( 16 );
228 
229  cugar::Timer timer;
230  timer.start();
231 
232  PTLoopStats stats;
233 
234  // update the direct-lighting VTLs Reinforcement-Learning tables
235  if (m_options.nee_type == NEE_ALGORITHM_RL)
236  update_vtls_rl( instance );
237 
238  // setup the samples for this frame
239  m_sequence.set_instance(instance);
240 
241  // use our PTLib pathtracer
242  {
244  // use the Reinforcement-Learning direct-lighting sampler
245  if (m_options.nee_type == NEE_ALGORITHM_RL)
246  {
247  // initialize the path-tracing context
248  PathTracingContext<DirectLightingRL> context;
249  context.options = m_options;
250  context.in_bounce = 0;
251  context.in_queue = in_queue;
252  context.scatter_queue = scatter_queue;
253  context.shadow_queue = shadow_queue;
254  context.sequence = m_sequence.view();
255  context.frame_weight = 1.0f / float(renderer_view.instance + 1);
256  context.device_timers = device_timers;
257  context.bbox = m_bbox;
258  context.dl = DirectLightingRL(
259  view( *m_vtls_rl ),
260  m_mesh_vtls->view() );
261 
262  // instantiate the actual path tracing loop
263  path_trace_loop( context, vertex_processor, renderer, renderer_view, stats );
264  }
265  else // use the regular mesh emitter direct-lighting sampler
266  {
267  // select which instantiation of the mesh light to use (VPLs or the plain mesh)
268  MeshLight mesh_light = m_options.nee_type == NEE_ALGORITHM_VPL ? renderer_view.mesh_vpls : renderer_view.mesh_light;
269 
270  // initialize the path-tracing context
271  PathTracingContext<DirectLightingMesh> context;
272  context.options = m_options;
273  context.in_bounce = 0;
274  context.in_queue = in_queue;
275  context.scatter_queue = scatter_queue;
276  context.shadow_queue = shadow_queue;
277  context.sequence = m_sequence.view();
278  context.frame_weight = 1.0f / float(renderer_view.instance + 1);
279  context.device_timers = device_timers;
280  context.bbox = m_bbox;
281  context.dl = DirectLightingMesh( mesh_light );
282 
283  // instantiate the actual path tracing loop
284  path_trace_loop( context, vertex_processor, renderer, renderer_view, stats );
285  }
287  }
288  timer.stop();
289  const float time = timer.seconds();
290  // clear the global timer at instance zero
291  if (instance == 0)
292  m_time = time;
293  else
294  m_time += time;
295 
296  fprintf(stderr, "\r %.1fs (%.1fms = rt[%.1fms + %.1fms + %.1fms] + shade[%.1fms + %.1fms] - %uK cells) ",
297  m_time,
298  time * 1000.0f,
299  stats.primary_rt_time * 1000.0f,
300  stats.path_rt_time * 1000.0f,
301  stats.shadow_rt_time * 1000.0f,
302  stats.path_shade_time * 1000.0f,
303  stats.shadow_shade_time * 1000.0f,
304  m_options.nee_type == NEE_ALGORITHM_RL ? m_vtls_rl->size() / 1000 : 0);
305 
306 #if defined(DEVICE_TIMING) && DEVICE_TIMING
307  if (instance % 64 == 0)
308  print_timer_stats( device_timers, stats );
309 #endif
310 
311  if (instance) // skip the first frame
312  {
313  m_stats.primary_rt_time += stats.primary_rt_time;
314  m_stats.path_rt_time += stats.path_rt_time;
315  m_stats.shadow_rt_time += stats.shadow_rt_time;
316  m_stats.path_shade_time += stats.path_shade_time;
317  m_stats.shadow_shade_time += stats.shadow_shade_time;
318  }
319  renderer.update_variances( instance );
320 }
