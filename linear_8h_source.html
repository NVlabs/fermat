42 template <typename T, int N>
43 Vector<T,N> scale_factor(const Matrix<T,N,N>& A)
44 {
45  int i,j;
46  Vector<T,N> S;
47 
48  for (i = 0; i < N; ++i)
49  {
50  S[i] = fabs(A[i][0]);
51  for (j = 1; j < N; ++j)
52  {
53  if (S[i] < fabs(A[i][j]))
54  S[i] = fabs(A[i][j]);
55  }
56  }
57  return S;
58 }

62 template <typename T, int N>
63 bool pivot_partial(Matrix<T,N,N>& A, const Vector<T,N>& S, Vector<T,N>& B)
64 {
65  int i,j;
66  T temp;
67  for(j = 0; j < N; ++j)
68  {
69  for(i = j + 1; i < N; ++i)
70  {
71  if (S[i] == 0)
72  {
73  if(B[i] == 0)
74  return false; // System doesn´t have a unique solution
75  else
76  return false; // System is incosistent
77  }
78  if (fabs(A[i][j]/S[i]) > fabs(A[j][j]/S[j]))
79  {
80  std::swap( A[i], A[j] );
81  temp = B[i];
82  B[i] = B[j];
83  B[j] = temp;
84  }
85  }
86 
87  if (A[j][j] == 0)
88  return false; // System is singular
89  }
90  return true;
91 }

95 template <typename T, int N>
96 bool forward_elimination(Matrix<T,N,N>& A, Vector<T,N>& B)
97 {
98  int i,j,k;
99  T m;
100 
101  for (k = 0; k < N-1; ++k)
102  {
103  for (i = k + 1; i < N; ++i)
104  {
105  m = A[i][k] / A[k][k];
106  for (j = k + 1; j < N; ++j)
107  {
108  A[i][j] -= m * A[k][j];
109  if (i == j && A[i][j] == 0)
110  return false; // Singular system
111  }
112  B[i] -= m * B[k];
113  }
114  }
115  return true;
116 }

120 template <typename T, int N>
121 Vector<T,N> back_substitution(Matrix<T,N,N>& A, Vector<T,N>& B)
122 {
123  int i,j;
124  T sum;
125  Vector<T,N> X;
126  X[N-1] = B[N-1]/A[N-1][N-1];
127  for (i = N - 2; i >= 0; --i)
128  {
129  sum = 0;
130  for (j = i + 1; j < N; ++j)
131  sum += A[i][j] * X[j];
132  X[i] = (B[i] - sum) / A[i][i];
133  }
134  return X;
135 }
