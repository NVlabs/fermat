34 {
35  return Vector3f(
36  cosf(uv[0])*sinf(uv[1]),

40 // computes the spherical coordinates of a 3d point
41 inline CUGAR_HOST CUGAR_DEVICE Vector2f to_spherical_coords(const Vector3f& vec)
42 {
43  // to spherical in [0,2pi] (phi) and [0,pi] (theta).
44  float theta, phi;
45  if (fabsf(vec[2]) >= 1.0f - 1.0e-5f)
46  phi = 0.0f;
47  else {
48  phi = atan2f( vec[1], vec[0] );
49  phi = phi<0.0f ? phi+2.0f*M_PIf : phi;
50  }
51  theta = acosf( vec[2] );
52  return Vector2f(phi,theta);
53 }

55 // seedx, seedy is point on [0,1]^2. x, y is point on radius 1 disk
56 inline CUGAR_HOST CUGAR_DEVICE Vector2f square_to_unit_disk(const Vector2f seed)
57 {
58  float phi, r;
59 
60  float a = 2*seed[0] - 1; // (a,b) is now on [-1,1]^2
61  float b = 2*seed[1] - 1;
62 
63  if (a > -b) { // region 1 or 2
64  if (a > b) { // region 1, also |a| > |b|
65  r = a;
66  phi = (M_PIf/4 ) * (b/a);
67  }
68  else { // region 2, also |b| > |a|
69  r = b;
70  phi = (M_PIf/4) * (2 - (a/b));
71  }
72  }
73  else { // region 3 or 4
74  if (a < b) { // region 3, also |a| >= |b|, a != 0
75  r = -a;
76  phi = (M_PIf/4) * (4 + (b/a));
77  }
78  else { // region 4, |b| >= |a|, but a==0 and b==0 could occur.
79  r = -b;
80  phi = b != 0 ? (M_PIf/4) * (6 - (a/b)) : 0;
81  }
82  }
83 
84  return Vector2f(
85  r * cosf(phi),
86  r * sinf(phi) );
87 }

89 // diskx, disky is point on radius 1 disk. x, y is point on [0,1]^2
90 inline CUGAR_HOST CUGAR_DEVICE Vector2f unit_disk_to_square(const Vector2f disk)
91 {
92  float r = sqrtf( disk[0]*disk[0] + disk[1]*disk[1] );
93  float phi = atan2f( disk[1], disk[0] );
94  float a, b;
95  if (phi < -M_PIf/4) phi += 2*M_PIf; // in range [-pi/4,7pi/4]
96  if ( phi < M_PIf/4) { // region 1
97  a = r;
98  b = phi * a / (M_PIf/4);
99  }
100  else if ( phi < 3*M_PIf/4 ) { // region 2
101  b = r;
102  a = -(phi - M_PIf/2) * b / (M_PIf/4);
103  }
104  else if ( phi < 5*M_PIf/4 ) { // region 3
105  a = -r;
106  b = (phi - M_PIf) * a / (M_PIf/4);
107  }
108  else { // region 4
109  b = -r;
110  a = -(phi - 3*M_PIf/2) * b / (M_PIf/4);
111  }
112 
113  return Vector2f(
114  (a + 1) / 2,
115  (b + 1) / 2 );
116 }

161 // maps the unit square to the sphere with a uniform distribution
162 inline CUGAR_HOST CUGAR_DEVICE Vector3f uniform_square_to_sphere(const Vector2f& uv)
163 {
164  const float cosTheta = uv.y*2.0f - 1.0f;
165  const float sinTheta = fast_sqrt(max(1.0f - cosTheta*cosTheta,0.0f));
166  const float phi = uv.x * M_TWO_PIf;

169  fast_cos(phi)*sinTheta,
170  fast_sin(phi)*sinTheta,

187 // Assume normalized input on +Z hemisphere.
188 // Output is on [-1, 1].
189 inline CUGAR_HOST CUGAR_DEVICE Vector2f hemisphere_to_hemioct(Vector3f v)
190 {
191  // Project the hemisphere onto the hemi-octahedron,
192  // and then into the xy plane
193  Vector2f p = v.xy() * (1.0f / (fabsf(v.x) + fabsf(v.y) + v.z));
194  // Rotate and scale the center diamond to the unit square
195  return Vector2f(p.x + p.y, p.x - p.y);
196 }
197 inline CUGAR_HOST CUGAR_DEVICE Vector3f hemioct_to_hemisphere(Vector2f e)
198 {
199  // Rotate and scale the unit square back to the center diamond
200  Vector2f temp = Vector2f(e.x + e.y, e.x - e.y) * 0.5f;
201  Vector3f v = Vector3f(temp, 1.0f - fabsf(temp.x) - fabsf(temp.y));
202  return normalize(v);
203 }

211 // Assume normalized input. Output is on [-1, 1] for each component.
212 inline CUGAR_HOST CUGAR_DEVICE Vector2f sphere_to_oct(Vector3f v)
213 {
214  // Project the sphere onto the octahedron, and then onto the xy plane
215  Vector2f p = v.xy() * (1.0f / (abs(v.x) + abs(v.y) + abs(v.z)));
216  // Reflect the folds of the lower hemisphere over the diagonals
217  return (v.z <= 0.0) ? ((Vector2f(1.0f) - abs(p.yx())) * signNotZero(p)) : p;
218 }
219 inline CUGAR_HOST CUGAR_DEVICE Vector3f oct_to_sphere(Vector2f e)
220 {
221  Vector3f v = Vector3f(e.xy(), 1.0f - abs(e.x) - abs(e.y));
222  if (v.z < 0) v.xy() = (Vector2f(1.0f) - abs(v.yx())) * signNotZero(v.xy());
223  return normalize(v);
224 }
