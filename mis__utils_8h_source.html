51 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
52 float cutoff_heuristic(const float p1, const float p2, const float alpha = 0.1f)
53 {
54  const float p_max = fmaxf(p1, p2);
55  const float q1 = p1 < alpha * p_max ? 0.0f : p1;
56  const float q2 = p2 < alpha * p_max ? 0.0f : p2;
57  return q1 / (q1 + q2);
58 }

60 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
61 float soft_cutoff_factor(const float w, const float threshold = 0.05f)
62 {
63  const float t = fminf(w / threshold, 1.0f);
64 
65  return t * t * (3.0f - 2.0f * t);
66 }

71 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
72 float soft_cutoff_heuristic(const float p1, const float p2, const float threshold = 0.05f)
73 {
74  // a balance heuristic which applies soft-clamping to low relative values of p_i
75  const float q1 = p1 * soft_cutoff_factor(p1 / (p1 + p2), threshold);
76  const float q2 = p2 * soft_cutoff_factor(p2 / (p1 + p2), threshold);
77  return q1 / (q1 + q2);
78 }

83 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
84 float biased_downweighting_heuristic(const float p1, const float p2, const float threshold = 0.05f)
85 {
86  // a balance heuristic which applies soft-clamping to low relative values of p_i
87  const float w = p1 / (p1 + p2);
88  return w * soft_cutoff_factor(w, threshold);

104 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
105 float mis_heuristic(const float p1, const float p2, const MisHeuristicType selector)
106 {
107  if (selector == BALANCE_HEURISTIC)
108  return balance_heuristic(p1, p2);
109  else if (selector == POWER_HEURISTIC)
110  return power_heuristic(p1, p2);
111  else if (selector == CUTOFF_HEURISTIC)
112  return cutoff_heuristic(p1, p2);
113  else if (selector == SOFT_CUTOFF_HEURISTIC)
114  return soft_cutoff_heuristic(p1, p2);
115  else
116  return biased_downweighting_heuristic(p1, p2);
117 }
