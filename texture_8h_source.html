252  void generate_mips()
253  {
254  uint32 level = 1;
255 
256  uint32 l_res_x = res_x/2;
257  uint32 l_res_y = res_y/2;
258  while (l_res_x >= 1 && l_res_y >= 1)
259  {
260  levels[level] = TexturePtr(new TextureType);
261  levels[level]->resize(l_res_x, l_res_y);
262 
263  downsample(levels[level - 1], levels[level]);
264 
265  level++;
266  l_res_x /= 2;
267  l_res_y /= 2;
268  }
269  }

271  static void downsample(TexturePtr src, TexturePtr dst)
272  {
273  TextureView src_view = src->view();
274  TextureView dst_view = dst->view();
275 
276  for (uint32 y = 0; y < dst_view.res_y; ++y)
277  {
278  for (uint32 x = 0; x < dst_view.res_x; ++x)
279  {
280  cugar::Vector4f t(0.0f);
281  for (uint32 j = 0; j < 2; ++j)
282  for (uint32 i = 0; i < 2; ++i)
283  t += cugar::Vector4f( src_view(x * 2 + i, y * 2 + j) );
284 
285  dst_view(x, y) = t / 4.0f;
286  }
287  }
288  }

306 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
307 cugar::Vector4f texture_lookup(float4 st, const TextureReference texture_ref, const MipMapView* textures, const float4 default_value)
308 {
309  if (texture_ref.texture == uint32(-1) || textures[texture_ref.texture].n_levels == 0)
310  return default_value;
311 
312  st.x *= texture_ref.scaling.x;
313  st.y *= texture_ref.scaling.y;
314 
315  st.x = cugar::mod(st.x, 1.0f);
316  st.y = cugar::mod(st.y, 1.0f);
317 
318  const MipMapView texture = textures[texture_ref.texture];
319  if (texture.n_levels > 0)
320  {
321  const uint32 x = cugar::min(uint32(st.x * texture.res_x), texture.res_x - 1);
322  const uint32 y = cugar::min(uint32(st.y * texture.res_y), texture.res_y - 1);
323  return texture(x, y, 0); // access LOD 0
324  }
325  else
326  return default_value;
327 }
