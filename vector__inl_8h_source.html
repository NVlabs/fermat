354 template <typename T>
355 CUGAR_HOST_DEVICE CUGAR_FORCEINLINE
356 Vector<T, 3> refract(const Vector<T, 3> I, const Vector<T, 3> N, const float eta)
357 {
358  const float N_dot_I = dot(N, I);
359  const float cos_theta_t2 = 1.f - eta * eta * (1.f - N_dot_I * N_dot_I);
360  if (cos_theta_t2 < 0.f)
361  return reflect(I,N); // total internal reflection
362  else
363  return eta * I - (eta * N_dot_I + sqrtf(cos_theta_t2)) * N;
364 }

377 template <typename T>
378 CUGAR_HOST_DEVICE
379 Vector<T, 3> orthogonal(const Vector<T, 3> v)
380 {
381  if (v[0] * v[0] < v[1] * v[1])
382  {
383  if (v[0] * v[0] < v[2] * v[2])
384  {
385  // r = -cross( v, (1,0,0) )
386  return Vector<T, 3>(0.0f, -v[2], v[1]);
387  }
388  else
389  {
390  // r = -cross( v, (0,0,1) )
391  return Vector<T, 3>(-v[1], v[0], 0.0);
392  }
393  }
394  else
395  {
396  if (v[1] * v[1] < v[2] * v[2])
397  {
398  // r = -cross( v, (0,1,0) )
399  return Vector<T, 3>(v[2], 0.0, -v[0]);
400  }
401  else
402  {
403  // r = -cross( v, (0,0,1) )
404  return Vector<T, 3>(-v[1], v[0], 0.0);
